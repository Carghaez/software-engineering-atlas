# Radix sort (LSD): stable counting-sort by each digit, least significant first

function radix_sort(a, base = 10):
    max_val = max(a)
    exp = 1
    while max_val / exp > 0:               # one pass per digit position
        counting_sort_by_digit(a, exp, base)
        exp = exp * base
    # once the most significant digit is processed, a is fully sorted
