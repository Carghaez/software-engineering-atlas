# Merge sort: split in half, sort each, merge the sorted halves (stable, O(n log n))

function merge_sort(a):
    if length(a) <= 1:
        return a
    mid   = length(a) / 2
    left  = merge_sort(a[0 .. mid])
    right = merge_sort(a[mid .. end])
    return merge(left, right)

function merge(left, right):
    out = []
    i = 0; j = 0
    while i < length(left) and j < length(right):
        if left[i] <= right[j]:           # <= keeps equal keys stable
            append left[i] to out; i = i + 1
        else:
            append right[j] to out; j = j + 1
    append remaining of left  (from i) to out
    append remaining of right (from j) to out
    return out
