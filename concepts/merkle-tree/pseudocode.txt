# Merkle tree: a binary hash tree; the root hash commits to every leaf

function merkle_root(leaves):
    level = [ hash(leaf) for leaf in leaves ]
    while length(level) > 1:
        if length(level) is odd:
            duplicate the last node           # pad to an even count
        next = []
        for i in steps of 2:
            next.append( hash(level[i] concat level[i+1]) )
        level = next
    return level[0]                           # the Merkle root
